found!
xkb_keymap {
    xkb_keycodes  { include "evdev"    };
    xkb_types     { include "complete"    };
    xkb_compat    { include "complete"    };
    xkb_symbols   { include "pc(pc105)+us(intl)"    };
    xkb_geometry  { include "pc(pc105)"    };
};

the culpit was xkb_keycodes  { include "evdev"    };
ufff did annoyed me a lot, but now this is working
