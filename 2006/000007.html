Hi 

During install - stage it says error:
/root/.sorcery/grimoire/Linux-PAM: line 42: move: command not found

Now looking augur details Linux-PAM and its build() function:

build() {
  unpack_docs() {
    mkdir docs          &&
    cd    docs          &&
    unpack ${SOURCE[1]} &&
    cd    -
  }

  move_la_files()  {
    for              file  in  /lib{,64}/libpam{,c_misc}.la;  do
      if    [   -f  $file  ]
      then  mv  -v  $file  /usr/$file
      fi
    done
  }

  export HAVE_SGML2LATEX=no
  sed  -i  "s: doc examples::g"      Makefile.in

  for   DIR  in  $(  find  .  -type d  )
  do    touch  $DIR/.ignore_age
  done

  unpack_docs    &&
  default_build  &&
  move la_files
}


Okay, it is typo: move la_files, should it be move_la_files ?-)
