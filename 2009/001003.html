ie something that do

# augur forgetopt xinetd

forgetopt () {
    for spell in $(grep -l ${1} /etc/sorcery/log/opt.on); do
         sed -i "s:${1}::g" ${spell};
         echo "${1}" >> ${spell/opt.on/opt.off}
    done
}
